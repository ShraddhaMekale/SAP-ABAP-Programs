# PROGRAM FOR ALL SELECTED ROWS FROM SUPPLIER MASTER TABLE (LFA1)

REPORT ZSQL_KNA1_1.

DATA : WA TYPE LFA1,
      ITAB LIKE TABLE OF LFA1.

START-OF-SELECTION.

SELECT MANDT LIFNR LAND1 NAME1 PSTL2 FROM LFA1
  UP TO 100 ROWS
  INTO CORRESPONDING FIELDS OF TABLE ITAB.

  LOOP AT ITAB INTO WA.

    WRITE : / WA-MANDT,
              WA-LIFNR,
              WA-LAND1,
              WA-NAME1,
              WA-PSTL2.

    ENDLOOP.
    
    WRITE : / SY-DBCNT.
