# PROGRAM FOR ARITHMATIC OBLIGATORY

REPORT ZARITHMATIC_OBLIGATORY.

PARAMETERS : A TYPE I OBLIGATORY,
             B TYPE I DEFAULT 10,
             D TYPE C LENGTH 50 VISIBLE LENGTH 20.

DATA :  C TYPE D,
        LENGTH TYPE I,
        REV TYPE STRING.

LENGTH = STRLEN(D).
REV = REVERSE(D).

C = SY_DATUM + 1.

WRITE : '15 TEXT - 001',C.
NEW-LINE.
WRITE : '15 TEXT - 002',C.
WRITE :SY-UNAME , SY-DATUM, SY-REPID.
WRITE : /' THE LENGTH OF A STRING IS', LENGTH.
WRITE : /' THE REVERSE OF A STRING IS', REV.
