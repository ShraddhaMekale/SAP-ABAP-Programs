# PROGRAM FOR CASE STATEMENT

REPORT Z_CASESTATEMENT.

PARAMETERS: NUM TYPE I.

START-OF-SELECTION.

CASE NUM.

  WHEN '1'.
    MESSAGE : ' THE NO. IS 1 ' TYPE 'W'.

  WHEN '2'.
    WRITE : ' THE NO. IS 2' .

  WHEN '3'.
    WRITE : ' THE NO. IS 3'.

  WHEN '4'.
    WRITE : ' THE NO. IS 4' .

  WHEN '5'.
    WRITE : ' THE NO. IS 5'.

  WHEN OTHERS .
    MESSAGE : ' ENTER THE NO. ONLY BETWEEN 1 TO 5 ' TYPE 'E'.

ENDCASE.
