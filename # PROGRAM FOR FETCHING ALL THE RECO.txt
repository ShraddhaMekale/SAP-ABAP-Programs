# PROGRAM FOR FETCHING ALL THE RECORDS FROM MARA TABLE


REPORT Z_MARA4.

DATA : WA TYPE MARA.

PARAMETERS : PA_MATNR TYPE C LENGTH 40.

START-OF-SELECTION.

SELECT SINGLE MATNR
              ERSDA
              ERNAM
              DATAB
              LIQDT

  FROM MARA INTO CORRESPONDING FIELDS OF WA
  WHERE MATNR = PA_MATNR.

  WRITE : WA-MATNR,
          WA-ERSDA,
          WA-ERNAM,
          WA-DATAB,
          WA-LIQDT.

  IF SY-SUBRC = 0.

    MESSAGE : ' SUCCESSFULLY FETCHED THE RECORD ' TYPE 'S'.

    WRITE : WA-MATNR,
            WA-ERSDA,
            WA-ERNAM,
            WA-DATAB,
            WA-LIQDT.
    ELSE.

      MESSAGE : ' NO RECORDS FOUND ' TYPE 'E'.

   ENDIF.
