# PROGRAM FOR SELECT MATERIAL NO ORDER BY ASCENDING


REPORT Z_SM2.
*
TABLES : MARA.

SELECT-OPTIONS : S_MATNR FOR MARA-MATNR,
                 S_ERSDA FOR MARA-ERSDA.

DATA : WA TYPE MARA,
      ITAB LIKE TABLE OF MARA.

START-OF-SELECTION.

SELECT MATNR ERSDA ERNAM FROM MARA
  INTO CORRESPONDING FIELDS OF TABLE ITAB
  WHERE
       MATNR IN S_MATNR
  AND
       ERSDA IN S_ERSDA

  ORDER BY MATNR ASCENDING.
  IF SY-SUBRC = 0 .

    WRITE: / SY-DBCNT.

    LOOP AT ITAB INTO WA.

      WRITE : / WA-MATNR,
                WA-ERSDA,
                WA-ERNAM.

   ENDLOOP.
   ENDIF.
