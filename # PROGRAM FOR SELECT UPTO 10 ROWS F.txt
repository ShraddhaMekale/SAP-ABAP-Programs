# PROGRAM FOR SELECT UPTO 10 ROWS FROM MARA TABLE

REPORT ZSQL_1.

DATA : WA TYPE MARA,
       ITAB LIKE TABLE OF MARA.

START-OF-SELECTION.

SELECT MATNR ERSDA ERNAM LAEDA AENAM PSTAT MTART FROM MARA UP TO 10 ROWS
  INTO CORRESPONDING FIELDS OF TABLE ITAB.

  WRITE : / SY-DBCNT.

  LOOP AT ITAB INTO WA.

    WRITE : / WA-MATNR,
              WA-ERSDA,
              WA-ERNAM,
              WA-LAEDA,
              WA-AENAM,
              WA-PSTAT,
              WA-MTART.

 ENDLOOP.
 END-OF-SELECTION.
