REPORT Z_SM2.

*
TABLES: MARA.

SELECT-OPTIONS S_MATNR FOR MARA-MATNR.

DATA: WA TYPE MARA,
      ITAB TYPE TABLE OF MARA.

START-OF-SELECTION.

SELECT MATNR ERSDA ERNAM
  FROM MARA
  INTO CORRESPONDING FIELDS OF TABLE ITAB

  WHERE MATNR IN S_MATNR
  ORDER BY MATNR ASCENDING.

  IF SY-SUBRC = 0.

  WRITE: / SY-DBCNT.

  LOOP AT ITAB INTO WA.

  WRITE: / WA-MATNR,
           WA-ERSDA,
           WA-ERNAM.

   ENDLOOP.
   ENDIF.
