REPORT ZJOIN_SHRI_TO_HERO.
TABLES : ZSHRI_DATA, ZHERO_HONDAID.
TYPES:BEGIN OF TY_FINAL,
      STUDENTS_APP_ID TYPE ZSHRI_DATA-STUDENTS_APP_ID,
      STUDENT_NAME TYPE ZSHRI_DATA-STUDENT_NAME,
      EMPLOYEE_ID TYPE ZHERO_HONDAID-EMPLOYEE_ID,
      EMPLOYEE_NAME TYPE ZHERO_HONDAID-EMPLOYEE_NAME,
      END  OF TY_FINAL.
*  SELECT-OPTIONS:S_STUD FOR ZSHRI_DATA-STUDENTS_APP_ID.
  DATA: WA TYPE TY_FINAL,
        ITAB LIKE TABLE OF WA.
  START-OF-SELECTION.
  SELECT A~STUDENTS_APP_ID A~STUDENT_NAME  B~EMPLOYEE_ID B~EMPLOYEE_NAME
         FROM ZSHRI_DATA AS A INNER JOIN ZHERO_HONDAID AS B ON A~STUDENTS_APP_ID = B~EMPLOYEE_ID
    INTO TABLE ITAB.

    WRITE : SY-DBCNT.
  LOOP AT ITAB INTO WA.
    WRITE: / WA-STUDENTS_APP_ID, WA-STUDENT_NAME, WA-EMPLOYEE_ID, WA-EMPLOYEE_NAME.
    ENDLOOP.
