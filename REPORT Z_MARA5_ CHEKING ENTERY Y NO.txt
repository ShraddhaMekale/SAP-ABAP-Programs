REPORT Z_MARA5.
DATA:WA TYPE MARA.
PARAMETERS:PA_MATNR TYPE C LENGTH 40.
START-OF-SELECTION.
SELECT SINGLE MATNR
              ERSDA
              ERNAM
              DATAB
              LIQDT
 FROM MARA INTO CORRESPONDING FIELDS OF WA
  WHERE MATNR = PA_MATNR.

  WRITE: WA-MATNR,
         WA-ERSDA,
         WA-ERNAM,
         WA-DATAB,
         WA-LIQDT.
